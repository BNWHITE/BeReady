React & Hooks

useState (Variable d'état)
import React, { useState } from 'react';

function Compteur() {
    const [count, setCount] = useState(0); // Init à 0

    return (
        <button onClick={() => setCount(count + 1)}>
            Clique : {count}
        </button>
    );
}
useEffect (Cycle de vie)
useEffect(() => {
    console.log("Composant monté");
    
    // Cleanup function (équivalent componentWillUnmount)
    return () => {
        console.log("Composant démonté");
    };
}, []); // [] = S'exécute une seule fois

Langage C

Gestion Mémoire : En C, si tu fais un malloc, tu DOIS faire un free. Sinon : fuite de mémoire (Memory Leak).
Pointeurs & Allocation Dynamique
#include <stdio.h>
#include <stdlib.h>

int main() {
    // Allocation d'un tableau de 10 entiers
    int *tab = (int*) malloc(10 * sizeof(int));

    if (tab == NULL) return 1; // Vérif erreur

    tab[0] = 42;
    
    // Libération obligatoire !
    free(tab);
    return 0;
}

OCaml

Fonction Récursive (Factorielle)
let rec fact n =
  if n = 0 then 1
  else n * fact (n - 1);;
  
(* Appel *)
fact 5;; (* Renvoie 120 *)
Pattern Matching (Listes)
let rec longueur liste =
  match liste with
  | [] -> 0          (* Cas de base : vide *)
  | tete::queue -> 1 + longueur queue;;
